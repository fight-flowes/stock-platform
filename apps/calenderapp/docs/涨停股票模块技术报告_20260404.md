## 一、项目背景

### 1.1 现有系统

**日历事件管理系统 (calenderappv4.1)**

| 模块 | 技术栈 | 功能 |
|------|--------|------|
| 后端 | Flask + Flask-RESTX + PostgreSQL + SQLAlchemy | REST API、数据持久化 |
| 前端 | Vue3 + Vite + Element Plus + FullCalendar | 日历视图、事件管理 |
| 数据 | PostgreSQL (schema: sc) | events、stocks、calendar_days |

**现有功能**：
- 日历事件管理（财报发布、股东大会等）
- 股票信息管理（代码、名称、交易所）
- 农历节假日显示
- CSV 批量导入导出

### 1.2 需求来源

**外部报告与数据链路已实现能力**：

| 脚本 | 功能 | 数据源 |
|------|------|--------|
| `fetch_limit_list.py` | 涨停股列表获取、ST过滤、连板统计 | Tushare limit_list_d |
| `fetch_lhb.py` | 龙虎榜分析、机构/游资/北向资金判断 | Tushare top_inst |

| `fetch_news_sina.py` | 个股专属新闻 | 新浪财经爬虫 |
| `fetch_company_info.py` | 公司基本信息 | Tushare stock_basic |

**核心工作流**：
```
获取涨停数据 → 公司信息 → 搜索新闻 → 分析资金 → 综合研判
```

## 二、功能设计

### 2.1 数据模型

#### 新增表：`sc.limit_up_stocks`

```python
class LimitUpStock(Base):
    __tablename__ = "limit_up_stocks"
    __table_args__ = {"schema": "sc"}

    # === 基本信息 ===
    id: int                  # 主键（自增）
    stock_code: str          # 股票代码（如 002361.SZ）
    stock_name: str          # 股票名称
    limit_up_date: date      # 涨停日期（YYYY-MM-DD）

    # === 涨停特征 ===
    consecutive_days: int    # 连板数（1=首板，2=二连板，3=三连板...）
    limit_up_type: str       # 涨停类型：first_board/multi_board/rebound
    seal_amount: float       # 封单金额（万元）
    seal_ratio: float        # 封单比（封单量/流通市值）
    turnover_rate: float     # 换手率（%）

    # === 时间信息 ===
    first_limit_time: time   # 首次涨停时间（如 09:35:00）
    last_limit_time: time    # 最后涨停时间
    open_count: int          # 开板次数（0=一字板，>0=曾开板）

    # === 板块概念 ===
    industry: str            # 所属行业
    concept_tags: JSONB      # 概念标签列表（如 ["航天", "军工"]）

    # === 资金动向（来自龙虎榜）===
    institution_net_buy: float   # 机构净买入（万元）
    hot_money_net_buy: float     # 游资净买入（万元）
    north_net_buy: float         # 北向资金净买入（万元）
    total_net_buy: float         # 总净买入（万元）

    # === 涨停原因 ===
    reason_category: str     # 原因分类：concept/fund/announcement/technical/event
    reason_detail: text      # 详细说明

    # === 强度评级 ===
    strength_level: int      # 强度等级（1-5星）
    strength_score: float    # 强度评分（综合计算）

    # === 龙头判断 ===
    is_dragon_head: bool     # 是否为空间龙头
    dragon_rank: int         # 龙头排名（当日连板高度排名）

    # === 元数据 ===
    source: str              # 数据来源（tushare/manual）
    created_at: datetime     # 创建时间
    updated_at: datetime     # 更新时间
```

#### 表关系

```
stocks (已有) ←── stock_code ──→ limit_up_stocks (新增)
                                        │
                                        ↓
                              events.stock_list (关联)
```

- `limit_up_stocks.stock_code` → `stocks.code`（外键可选）
- 涨停股可作为 `CalendarEvent.stock_list` 成员

**服务层 (limit_up_service.py)**：

```python
from datetime import date
from typing import Dict, List, Optional
from sqlalchemy import and_, func, select
from app.db import session_scope
from app.models.limit_up_stock import LimitUpStock
from app.models.stock import Stock

class LimitUpService:
    @staticmethod
    def list_limit_ups(
        page: int,
        page_size: int,
        start_date: Optional[str] = None,
        end_date: Optional[str] = None,
        consecutive_min: Optional[int] = None,
        industry: Optional[str] = None,
        concept: Optional[str] = None,
        strength_min: Optional[int] = None,
        is_dragon_head: Optional[bool] = None,
        q: Optional[str] = None
    ) -> Dict:
        """涨停股列表查询"""
        with session_scope() as session:
            query = select(LimitUpStock)
            
            where = []
            if start_date:
                where.append(LimitUpStock.limit_up_date >= date.fromisoformat(start_date))
            if end_date:
                where.append(LimitUpStock.limit_up_date <= date.fromisoformat(end_date))
            if consecutive_min:
                where.append(LimitUpStock.consecutive_days >= consecutive_min)
            if industry:
                where.append(LimitUpStock.industry == industry)
            if concept:
                where.append(LimitUpStock.concept_tags.contains([concept]))
            if strength_min:
                where.append(LimitUpStock.strength_level >= strength_min)
            if is_dragon_head:
                where.append(LimitUpStock.is_dragon_head == True)
            if q:
                where.append(LimitUpStock.stock_name.ilike(f"%{q}%"))
            
            if where:
                query = query.where(and_(*where))
            
            # 统计总数
            total_query = query.with_only_columns(func.count(LimitUpStock.id))
            total = session.execute(total_query).scalar()
            
            # 分页查询
            query = query.order_by(
                LimitUpStock.limit_up_date.desc(),
                LimitUpStock.consecutive_days.desc(),
                LimitUpStock.strength_level.desc()
            ).offset((page - 1) * page_size).limit(page_size)
            
            items = session.execute(query).scalars().all()
            
            return {
                "items": [item.to_dict() for item in items],
                "total": total
            }

    @staticmethod
    def get_consecutive_rank(trade_date: str) -> List[Dict]:
        """连板榜排名"""
        with session_scope() as session:
            items = session.execute(
                select(LimitUpStock)
                .where(LimitUpStock.limit_up_date == date.fromisoformat(trade_date))
                .where(LimitUpStock.consecutive_days >= 2)
                .order_by(LimitUpStock.consecutive_days.desc())
            ).scalars().all()
            return [item.to_dict() for item in items]

    @staticmethod
    def get_statistics(start_date: str, end_date: str) -> Dict:
        """区间统计"""
        with session_scope() as session:
            sd = date.fromisoformat(start_date)
            ed = date.fromisoformat(end_date)
            
            # 总数
            total = session.execute(
                select(func.count(LimitUpStock.id))
                .where(and_(LimitUpStock.limit_up_date >= sd, LimitUpStock.limit_up_date <= ed))
            ).scalar()
            
            # 连板分布
            by_consecutive = session.execute(
                select(LimitUpStock.consecutive_days, func.count())
                .where(and_(LimitUpStock.limit_up_date >= sd, LimitUpStock.limit_up_date <= ed))
                .group_by(LimitUpStock.consecutive_days)
                .order_by(LimitUpStock.consecutive_days.desc())
            ).all()
            
            # 行业分布
            by_industry = session.execute(
                select(LimitUpStock.industry, func.count())
                .where(and_(LimitUpStock.limit_up_date >= sd, LimitUpStock.limit_up_date <= ed))
                .where(LimitUpStock.industry.isnot(None))
                .group_by(LimitUpStock.industry)
                .order_by(func.count().desc())
                .limit(10)
            ).all()
            
            return {
                "total": total,
                "by_consecutive": {str(k): v for k, v in by_consecutive},
                "by_industry": {k: v for k, v in by_industry}
            }

    @staticmethod
    def sync_from_tushare(trade_date: str) -> Dict:
        """从 Tushare 同步涨停数据"""
```

## 四、强度评级算法

### 4.1 评分维度

| 维度 | 权重 | 指标 |
|------|------|------|
| 封板时间 | 25% | 9:30-10:00(满分) → 14:00后(最低) |
| 连板高度 | 20% | 连板数越高分越高 |
| 封单金额 | 20% | 封单占成交额比例 |
| 开板次数 | 15% | 0次(一字板)最高分 |
| 资金动向 | 20% | 机构+游资净买入 |

### 4.2 计算公式

```python
def calculate_strength_score(limit_up: LimitUpStock) -> float:
    """计算涨停强度评分 (0-100)"""
    
    # 1. 封板时间评分 (0-25分)
    if limit_up.first_limit_time:
        time_str = str(limit_up.first_limit_time)
        if time_str < "100000":
            time_score = 25  # 10点前封板
        elif time_str < "110000":
            time_score = 20
        elif time_str < "130000":
            time_score = 15
        elif time_str < "140000":
            time_score = 10
        else:
            time_score = 5  # 下午封板
    else:
        time_score = 10
    
    # 2. 连板高度评分 (0-20分)
    consecutive = limit_up.consecutive_days
    if consecutive >= 5:
        consecutive_score = 20
    elif consecutive >= 4:
        consecutive_score = 18
    elif consecutive >= 3:
        consecutive_score = 15
    elif consecutive >= 2:
        consecutive_score = 12
    else:
        consecutive_score = 8
    
    # 3. 封单金额评分 (0-20分) - 相对成交额
    if limit_up.seal_amount and limit_up.turnover_rate:
        seal_ratio = limit_up.seal_amount / (limit_up.seal_amount * limit_up.turnover_rate / 100)
        seal_score = min(20, seal_ratio * 10)
    else:
        seal_score = 10
    
    # 4. 开板次数评分 (0-15分)
    open_count = limit_up.open_count
    if open_count == 0:
        open_score = 15  # 一字板
    elif open_count == 1:
        open_score = 12
    elif open_count == 2:
        open_score = 8
    else:
        open_score = 5
    
    # 5. 资金动向评分 (0-20分)
    net_buy = limit_up.institution_net_buy + limit_up.hot_money_net_buy
    if net_buy > 100000:  # 1亿以上
        fund_score = 20
    elif net_buy > 50000:  # 5000万以上
        fund_score = 15
    elif net_buy > 10000:  # 1000万以上
        fund_score = 10
    elif net_buy > 0:
        fund_score = 5
    else:
        fund_score = 0
    
    total_score = time_score + consecutive_score + seal_score + open_score + fund_score
    
    # 转换为星级 (1-5)
    if total_score >= 80:
        strength_level = 5
    elif total_score >= 60:
        strength_level = 4
    elif total_score >= 40:
        strength_level = 3
    elif total_score >= 20:
        strength_level = 2
    else:
        strength_level = 1
    
    return total_score, strength_level
```
